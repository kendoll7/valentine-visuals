def consolidate_columns(participants_df):
    # Create an empty list to store the consolidated values
    consolidated_values = []

    # Iterate over each value in the DataFrame
    for index, row in participants_df.iterrows():
        for value in row:
            # Append each value to the consolidated values list
            consolidated_values.append(value)

    # Create a new DataFrame with the consolidated values
    consolidated_df = pd.DataFrame({'Consolidated': consolidated_values})

    return consolidated_df

all_words_lower = [word.lower() for sublist in words for word in sublist]

# Count the frequency of each word

# Convert the Counter object to a dictionary for easier access
word_counts_dict = dict(word_counts)

# Convert dictionary to DataFrame
word_counts_df = pd.DataFrame.from_dict(word_counts_dict, orient='index', columns=['Count'])

def remove_punctuation(text):
    # Define a string of punctuation characters
    punctuation = string.punctuation
    # Replace punctuation characters with an empty string
    return text.translate(str.maketrans('', '', punctuation))

# Apply the function to the 'Words' column

sorted_word_counts_df['Word'] = sorted_word_counts_df['Word'].apply(remove_punctuation)
